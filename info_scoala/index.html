#include <iostream>
#include <string>
using namespace std;

int i,t=5;
bool corect=true;

int main() {

    string orar[5]={"fizica","romana","mate","chimie","informatica"};
    string ghiozdan[5];
    string verificare[5];

    cout<<"Orarul pentru azi:"<<endl;
    for(i=0;i<5;i++){
        cout<<"- "<<orar[i]<<endl;
    }

    cout<<"Introdu cartile in ghiozdan in ordinea corecta :"<<endl;
    for(i=0;i<5;i++){
        string input;
        cin>>input;
        ghiozdan[i]=input;
    }

    for(i=0;i<5;i++){
        verificare[i]=orar[i];
    }

    for(i=0;i<5;i++){
        t--;
        if (ghiozdan[t]!=verificare[i]){
            corect=false;
            break;
        }
    }

    if(corect){
        cout<<endl<<"Ai introdus cartile cum trebuie!";
    }else{
        cout<<endl<<"Nu ai introdus cartile cum trebuie!";
    }

    return 0;
}
